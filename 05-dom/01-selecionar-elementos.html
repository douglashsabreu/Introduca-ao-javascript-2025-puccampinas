Exemplo: Selecionando Elementos

Abra o Console (F12) para ver os logs!

// Adiciona elementos HTML para demonstração
document.body.innerHTML += '
Olá
Parágrafo 1
Parágrafo 2
';

console.log("=== SELECIONANDO ELEMENTOS ===");

let titulo = document.getElementById("titulo");
console.log("Por ID:", titulo);

let paragrafos = document.getElementsByClassName("texto");
console.log("Por classe:", paragrafos);

let primeiro = document.querySelector(".texto");
console.log("querySelector:", primeiro);

let todos = document.querySelectorAll(".texto");
console.log("querySelectorAll:", todos);
