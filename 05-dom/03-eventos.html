Exemplo: Eventos

Abra o Console (F12) para ver os logs!

document.body.innerHTML = '
Eventos em JavaScript
';

let btn1 = document.getElementById("btn1");
let btn2 = document.getElementById("btn2");
let resultado = document.getElementById("resultado");

btn1.addEventListener("click", function() {
    resultado.textContent = "Botão 1 clicado!";
    console.log("Botão 1 foi clicado");
});

btn2.addEventListener("click", () => {
    resultado.textContent = "Botão 2 clicado!";
    console.log("Botão 2 foi clicado");
});

console.log("Clique nos botões para ver eventos em ação!");
