let pessoa = {
    nome: "Maria",
    idade: 25,
    cidade: "São Paulo"
};

console.log("Objeto:", pessoa);
console.log("Nome:", pessoa.nome);
console.log("Idade:", pessoa["idade"]);

pessoa.idade = 26;

let carro = {
    marca: "Toyota",
    modelo: "Corolla",
    ano: 2023,
    descrever: function() {
        return `${this.marca} ${this.modelo} ${this.ano}`;
    }
};

console.log("\nCarro:", carro.descrever());
