Arrays de Objetos

let alunos = [
    { nome: "Ana", nota: 85 },
    { nome: "João", nota: 92 },
    { nome: "Maria", nota: 78 }
];

console.log("Todos os alunos:");
alunos.forEach(aluno => {
    console.log(`${aluno.nome}: ${aluno.nota}`);
});

let aprovados = alunos.filter(aluno => aluno.nota >= 80);
console.log("\nAprovados:", aprovados);
