Métodos de Arrays

let lista = [1, 2, 3];

console.log("Array inicial:", lista);

lista.push(4);
console.log("Depois de push(4):", lista);

lista.pop();
console.log("Depois de pop():", lista);

lista.unshift(0);
console.log("Depois de unshift(0):", lista);

lista.shift();
console.log("Depois de shift():", lista);
